let currentQuestion = 0; let answers = Array(questions.length).fill(null); let timeLeft = 600; // 10 menit dalam detik let timerInterval; // Simpan interval timer const timerElement = document.getElementById('timer'); const questionCard = document.getElementById('question-card'); const questionNumber = document.getElementById('question-number'); const questionText = document.getElementById('question-text'); const optionsContainer = document.getElementById('options'); const questionMap = document.getElementById('question-map'); const prevBtn = document.getElementById('prev-btn'); const nextBtn = document.getElementById('next-btn'); const finishBtn = document.getElementById('finish-btn'); function updateTimer() { const minutes = Math.floor(timeLeft / 60); const seconds = timeLeft % 60; timerElement.textContent = `Sisa Waktu: ${minutes}:${seconds < 10 ? '0' : ''}${seconds}`; if (timeLeft > 0) { timeLeft--; } else { finishTO(); } } function startTimer() { timerInterval = setInterval(updateTimer, 1000); } function stopTimer() { clearInterval(timerInterval); timerElement.style.display = 'none'; // Sembunyikan timer } function renderQuestion(index) { const question = questions[index]; questionNumber.textContent = `Soal ${index + 1}`; questionText.textContent = question.question; optionsContainer.innerHTML = ''; question.options.forEach((option, i) => { const button = document.createElement('button'); button.textContent = option; button.className = 'option'; if (answers[index] === i) button.classList.add('selected'); button.addEventListener('click', () => handleAnswer(i)); optionsContainer.appendChild(button); }); prevBtn.disabled = index === 0; nextBtn.disabled = index === questions.length - 1; } function handleAnswer(optionIndex) { answers[currentQuestion] = optionIndex; renderQuestion(currentQuestion); renderQuestionMap(); } function navigateQuestion(index) { currentQuestion = index; renderQuestion(index); } function renderQuestionMap() { questionMap.innerHTML = ''; questions.forEach((_, index) => { const button = document.createElement('button'); button.textContent = index + 1; button.className = 'question-box'; if (answers[index] !== null) button.classList.add('answered'); button.addEventListener('click', () => navigateQuestion(index)); questionMap.appendChild(button); }); } function finishTO() { stopTimer(); // Hentikan timer dan sembunyikan const resultContainer = document.getElementById('result'); const scoreContainer = document.getElementById('score'); const explanationsContainer = document.getElementById('explanations'); let score = 0; // Hitung skor questions.forEach((question, index) => { if (answers[index] === question.correct) { score++; } }); // Tampilkan skor scoreContainer.textContent = `Skor Anda: ${score} dari ${questions.length}`; // Tampilkan pembahasan explanationsContainer.innerHTML = ''; questions.forEach((question, index) => { const explanationDiv = document.createElement('div'); explanationDiv.className = 'explanation'; explanationDiv.innerHTML = `

Soal ${index + 1}: ${question.question}

Jawaban Anda: ${answers[index] !== null ? question.options[answers[index]] : 'Tidak dijawab'}

Jawaban Benar: ${question.options[question.correct]}

Pembahasan: ${question.explanation}

`; explanationsContainer.appendChild(explanationDiv); }); // Sembunyikan soal, tampilkan hasil document.querySelector('main').style.display = 'none'; resultContainer.style.display = 'block'; } prevBtn.addEventListener('click', () => { if (currentQuestion > 0) navigateQuestion(currentQuestion - 1); }); nextBtn.addEventListener('click', () => { if (currentQuestion < questions.length - 1) navigateQuestion(currentQuestion + 1); }); finishBtn.addEventListener('click', finishTO); renderQuestion(currentQuestion); renderQuestionMap(); startTimer();
